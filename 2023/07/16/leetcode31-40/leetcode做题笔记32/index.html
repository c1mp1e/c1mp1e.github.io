思路一：利用栈的特性解决
先将左括号入栈，若后面栈顶为右括号则使前一个左括号出栈，最后输出最后一个右括号和栈底左括号之间的差值

    #define MAX(a , b) ((a) > (b) ? (a) : (b))
    int longestValidParentheses(char * s){
        int len = strlen(s);
        int max = 0;
        int str[len+1];
        int top = -1;
        str[++top] = -1;
        for(int i = 0; i < len; i++)
        {
            if(s[i] == '(')
            {
                str[++top] = i;
            }
            if(s[i] == ')')
            {
                --top;
                if(top == -1)
                {
                    str[++top] = i;
                }
                else
                {
                    max = MAX(max , (i - str[top]));
                }
            }
        }
        return max;
    }

时间复杂度O(n)，空间复杂度O(n)
